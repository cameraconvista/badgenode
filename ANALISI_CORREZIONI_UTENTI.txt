// Gestione caricamento dipendenti
async function caricaUtenti() {
  try {
    const { data: utenti, error } = await window.supabase
      .from('utenti')
      .select('*')
      .order('pin', { ascending: true });
    
    if (error) throw error;
    
    const tbody = document.getElementById('lista-dipendenti');
    tbody.innerHTML = '';
    
    utenti.forEach(utente => {
      const row = document.createElement('tr');
      row.innerHTML = `
        <td><button onclick="window.location.href='storico.html?pin=${utente.pin}'" 
            style="background:none;border:none;font-size:20px;cursor:pointer;" 
            title="Vedi storico">📊</button></td>
        <td>${utente.pin}</td>
        <td>${utente.nome}</td>
        <td>${utente.cognome}</td>
        <td>
          <button onclick="apriModalDettagli({pin: ${utente.pin}, nome: '${utente.nome}', cognome: '${utente.cognome}', email: '${utente.email || ''}', telefono: '${utente.telefono || ''}', descrizione_contratto: '${utente.descrizione_contratto || ''}', ore_contrattuali: ${utente.ore_contrattuali || 8.00}})" 
                  style="background:none;border:none;font-size:20px;cursor:pointer;" 
                  title="Modifica">✏️</button>
          <button onclick="archiviaUtente('${utente.pin}')" 
                  title="Archivia dipendente" 
                  style="background:none;border:none;font-size:20px;cursor:pointer;">📦</button>
          <button onclick="eliminaUtente('${utente.pin}', '${utente.nome}', '${utente.cognome}')" 
                  title="Elimina definitivamente" 
                  style="background:none;border:none;font-size:20px;cursor:pointer;color:#dc2626;">❌</button>
        </td>
      `;
      tbody.appendChild(row);
    });
    
  } catch (error) {
    console.error('Errore caricamento utenti:', error);
    alert('Errore nel caricamento dei dipendenti');
  }
}

// Gestione modal dettagli/modifica  
function apriModalDettagli(utente) {
  const modal = document.getElementById('modalDettagli');
  modal.style.display = 'flex';
  
  // Popola i campi
  document.getElementById('dettagli-nome').value = utente.nome;
  document.getElementById('dettagli-cognome').value = utente.cognome;
  document.getElementById('dettagli-email').value = utente.email || '';
  document.getElementById('dettagli-telefono').value = utente.telefono || '';
  document.getElementById('dettagli-descrizione-contratto').value = utente.descrizione_contratto || '';
  document.getElementById('dettagli-ore-contrattuali').value = utente.ore_contrattuali || 8.0;
  
  // Salva PIN per successivo aggiornamento
  modal.dataset.pin = utente.pin;
}

// Archiviazione dipendente
async function archiviaUtente(pin) {
  if (!confirm('Vuoi archiviare questo dipendente? Potrà essere ripristinato dalla pagina ex-dipendenti.')) return;
  
  try {
    // Ottieni dati utente
    const { data: utente, error: fetchError } = await window.supabase
      .from('utenti')
      .select('*')
      .eq('pin', pin)
      .single();
    
    if (fetchError) throw fetchError;
    
    // Inserisci in dipendenti_archiviati
    const { error: insertError } = await window.supabase
      .from('dipendenti_archiviati')
      .insert({
        pin_originale: utente.pin,
        nome: utente.nome,
        cognome: utente.cognome,
        data_archiviazione: new Date().toISOString()
      });
    
    if (insertError) throw insertError;
    
    // Rimuovi da utenti attivi
    const { error: deleteError } = await window.supabase
      .from('utenti')
      .delete()
      .eq('pin', pin);
    
    if (deleteError) throw deleteError;
    
    alert('✅ Dipendente archiviato con successo');
    caricaUtenti(); // Ricarica lista
    
  } catch (error) {
    console.error('Errore archiviazione:', error);
    alert('Errore durante l\'archiviazione: ' + error.message);
  }
}

// Eliminazione definitiva
async function eliminaUtente(pin, nome, cognome) {
  if (!confirm(`ATTENZIONE: Vuoi eliminare definitivamente ${nome} ${cognome}?\nQuesta azione NON può essere annullata.`)) return;
  
  try {
    const { error } = await window.supabase
      .from('utenti')  
      .delete()
      .eq('pin', pin);
    
    if (error) throw error;
    
    alert(`✅ ${nome} ${cognome} eliminato definitivamente`);
    caricaUtenti(); // Ricarica lista
    
  } catch (error) {
    console.error('Errore eliminazione:', error);
    alert('Errore durante l\'eliminazione: ' + error.message);
  }
}
